# but uses country scope so it can be shown before an election starts.

	add = modifier:country_election_cost_mult
	mult = trigger:empire_size

# Mirrors the first use of leader_election_support_cost.

}

# Five consecutive uses cost 1 + 1.5 + 2 + 2.5 + 3 = 10 base uses.
